


// List current directory
ls -al

// Create an new directory/folder
mkdir <dir-name>

// Change directory/folder
cd <dir-name>

// Change directory into the user root
cd ~

// Print out working directory
pwd

// Hard delete a folder or file
rm -rf <folder-or-filename>

// Initialize an empty local git repository
git init 

// Make a connection between local and remote repository
git remote add origin <github-url>

// Output the status of your working directory
git status

// Output a list of local branches
git branch
